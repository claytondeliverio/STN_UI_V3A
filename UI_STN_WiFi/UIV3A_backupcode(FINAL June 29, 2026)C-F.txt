static bool probeInternetTarget(const IPAddress &ip) {
  WiFiClient probeClient;
  bool ok = probeClient.connect(ip, 53, INTERNET_PROBE_TIMEOUT_MS);
  if (ok) {
    probeClient.stop();
  }
  return ok;
}
static void updateInternetProbeStatus() {
  if (!wifiOk()) {
    noInternetSinceMs = 0;
    lastInternetProbeMs = 0;
    return;
  }

  unsigned long now = millis();
  if (now - lastInternetProbeMs < INTERNET_PROBE_INTERVAL_MS) {
    return;
  }
  lastInternetProbeMs = now;

  bool internetOk = probeInternetTarget(IPAddress(1, 1, 1, 1)) || probeInternetTarget(IPAddress(8, 8, 8, 8));
  if (internetOk) {
    noInternetSinceMs = 0;
  } else if (noInternetSinceMs == 0) {
    noInternetSinceMs = now;
  }
}
static inline byte getNetworkErrorCode() {
  if (millis() < 3000) return 0;
  if (wifiConnectStartedAtMs != 0 && (millis() - wifiConnectStartedAtMs) < WIFI_CONNECT_ERROR_GRACE_MS) return 0;
  if (!wifiOk()) return 11;  // E11: not connected to WiFi
  unsigned long noInternetMs = noInternetSinceMs;
  if (noInternetMs != 0 && (millis() - noInternetMs) >= NO_INTERNET_E12_DELAY_MS) {
    return 12;  // E12: connected to WiFi but internet probe failed for 3 minutes
  }
  return 0;
}
static inline bool wifiTryingSavedCredentials() {

  return wifiConnectStartedAtMs != 0;

static inline uint32_t IRAM_ATTR displaySegmentGpioMask(uint8_t segs) {
  uint32_t mask = 0;
  if (segs & SEG_A) mask |= (1UL << 7);
  if (segs & SEG_B) mask |= (1UL << 6);
  if (segs & SEG_C) mask |= (1UL << 1);
  if (segs & SEG_D) mask |= (1UL << 0);
  if (segs & SEG_E) mask |= (1UL << 3);
  if (segs & SEG_F) mask |= (1UL << 2);
  if (segs & SEG_G) mask |= (1UL << 9);
  if (segs & SEG_DP) mask |= (1UL << 10);
  return mask;
}

static inline uint32_t IRAM_ATTR displayColumnGpioMask(uint8_t col) {
  switch (col) {
    case 0: return (1UL << 5);
    case 1: return (1UL << 4);
    case 2: return (1UL << 26);
    case 3: return (1UL << 25);
    case 4: return (1UL << 24);
    default: return 0;
  }
}

static void IRAM_ATTR displayRefreshIsr() {
  static uint8_t col = 0;
  static bool ledBankBlanked = false;

  REG_WRITE(GPIO_OUT_W1TS_REG, COLUMN_MASK);
  REG_WRITE(GPIO_OUT_W1TS_REG, SHARED_BTN_SEG_MASK | (1UL << 2) | (1UL << 3) | (1UL << 9) | (1UL << 10));

  if (col == DISPLAY_LED_BANK_COL && !ledBankBlanked) {
    ledBankBlanked = true;
    return;
  }

  uint8_t segs = displayRefreshFrame[col];
  if (segs != 0 && !(DISABLE_LED_BANK && col == 4)) {
    REG_WRITE(GPIO_OUT_W1TC_REG, displaySegmentGpioMask(segs));
    REG_WRITE(GPIO_OUT_W1TC_REG, displayColumnGpioMask(col));
  }

  col++;
  if (col >= DISPLAY_REFRESH_COLS) {
    col = 0;
  }
  ledBankBlanked = false;
}

static void displayCommitRefreshFrame(const uint8_t frame[DISPLAY_REFRESH_COLS]) {
  portENTER_CRITICAL(&displayRefreshMux);
  for (uint8_t i = 0; i < DISPLAY_REFRESH_COLS; i++) {
    displayRefreshFrame[i] = frame[i];
  }
  portEXIT_CRITICAL(&displayRefreshMux);
}

static void displayClearRefreshFrame() {
  uint8_t blank[DISPLAY_REFRESH_COLS] = {0, 0, 0, 0, 0};
  displayCommitRefreshFrame(blank);
}

static void displayStartRefreshTimer() {
  if (displayRefreshTimer != nullptr) return;
  displayClearRefreshFrame();
  displayRefreshTimer = timerBegin(DISPLAY_TIMER_HZ);
  if (displayRefreshTimer == nullptr) return;
  timerAttachInterrupt(displayRefreshTimer, displayRefreshIsr);
  timerWrite(displayRefreshTimer, 0);
  timerAlarm(displayRefreshTimer, DISPLAY_REFRESH_PERIOD_US, true, 0);
}

inline void displayBeginFrameOps() {
  if (!displayReplayEnabled) return;
  portENTER_CRITICAL(&displayOpsMux);
  displayBuildCount = 0;
  portEXIT_CRITICAL(&displayOpsMux);
}

inline void displayRecordOp(uint8_t type, int16_t arg = 0) {
  if (!displayReplayEnabled || displayReplayInProgress) return;
  portENTER_CRITICAL(&displayOpsMux);
  if (displayBuildCount < DISPLAY_OP_CAP) {
    displayBuildOps[displayBuildCount].type = type;
    displayBuildOps[displayBuildCount].arg = arg;
    displayBuildCount = displayBuildCount + 1;
  }
  portEXIT_CRITICAL(&displayOpsMux);
}

inline void displayCommitFrameOps() {
  if (!displayReplayEnabled) return;
  DisplayOp localOps[DISPLAY_OP_CAP];
  portENTER_CRITICAL(&displayOpsMux);
  uint8_t cnt = displayBuildCount;
  if (cnt > DISPLAY_OP_CAP) cnt = DISPLAY_OP_CAP;
  if (cnt > 0) {
    for (uint8_t i = 0; i < cnt; i++) {
      displayActiveOps[i] = displayBuildOps[i];
      localOps[i] = displayBuildOps[i];
    }
  }
  // Commit empty frames too, otherwise previous frame persists and blink "off" phase is invisible.
  displayActiveCount = cnt;
  portEXIT_CRITICAL(&displayOpsMux);

  uint8_t frame[DISPLAY_REFRESH_COLS];
  if (cnt == 0) {
    displayClearRefreshFrame();
  } else {
    displayRenderOpsToFrame(localOps, cnt, frame);
    displayCommitRefreshFrame(frame);
  }
}

inline void displayBlankFrameOps() {
  if (!displayReplayEnabled) return;
  portENTER_CRITICAL(&displayOpsMux);
  displayBuildCount = 0;
  displayActiveCount = 0;
  portEXIT_CRITICAL(&displayOpsMux);
  displayClearRefreshFrame();
}

inline byte debounceButtonState(byte rawState, ButtonDebounceState &state, unsigned long nowMs) {

void startPortalNonBlocking(bool wipeCredentials) {
  ledOff();
  wifiPortalCredentialsSaved = false;
  wifiPortalSavedAtMs = 0;
  wifiConnectStartedAtMs = 0;
  wifiPortalCloseAtMs = 0;

  // Only erase stored STA credentials when explicitly requested.
  WiFi.disconnect(true, wipeCredentials);

  if (wipeCredentials) {
    wm.resetSettings();
  }

  configureWifiManager();
  wm.startConfigPortal(wifiAPName.c_str());  // returns immediately
  portalActive = true;

  // reset reconnect timers so we don't instantly re-trigger logic
  disconnectSince = 0;
  lastKick = 0;
}

void stopPortalIfPossible() {
  wm.stopConfigPortal();
  portalActive = false;
}

void kickReconnectWindow() {
  unsigned long now = millis();

  if (disconnectSince == 0) disconnectSince = now;
  if (wifiConnectStartedAtMs == 0) wifiConnectStartedAtMs = now;

  if (now - lastKick >= RECONNECT_KICK_MS) {
    lastKick = now;
    WiFi.reconnect();  // or WiFi.begin(); if you prefer
  }

  if (now - disconnectSince >= RECONNECT_WINDOW_MS) {
    startPortalNonBlocking(false);  // timeout -> open portal (no wipe)
  }
}

void wifiService() {
  if (digitalRead(RESET_BTN) == LOW) {
    if (resetBtnPressedAt == 0) {
      resetBtnPressedAt = millis();
    }
    if (!resetBtnTriggered && millis() - resetBtnPressedAt >= RESET_BTN_LONG_PRESS_MS) {
      resetBtnTriggered = true;
      startPortalNonBlocking(true);
      return;
    }
  } else {
    resetBtnPressedAt = 0;
    resetBtnTriggered = false;
  }

  if (wifiOk()) {
    ledBlue();
    disconnectSince = 0;
    lastKick = 0;
    bootGraceHandled = true;
    wifiConnectStartedAtMs = 0;
    wifiPortalCredentialsSaved = false;
    updateInternetProbeStatus();

    if (portalActive) {
      if (wifiPortalCloseAtMs == 0) {
        wifiPortalCloseAtMs = millis() + WIFI_PORTAL_CLOSE_AFTER_CONNECT_MS;
      } else if (millis() >= wifiPortalCloseAtMs) {
        stopPortalIfPossible();
        wifiPortalCloseAtMs = 0;
      }
    }
    return;
  }

  ledOff();
  noInternetSinceMs = 0;
  lastInternetProbeMs = 0;

  if (portalActive) {
    unsigned long now = millis();
    if (now - lastPortalProcessMs >= WIFI_PORTAL_PROCESS_INTERVAL_MS) {
      lastPortalProcessMs = now;
      wm.process();  // keep portal alive but avoid hogging CPU
    }
    if (wifiPortalCredentialsSaved && (millis() - wifiPortalSavedAtMs) >= WIFI_PORTAL_CLOSE_AFTER_SAVE_MS) {
      if (wifiConnectStartedAtMs == 0) {
        WiFi.setAutoReconnect(true);
        WiFi.begin();
        wifiConnectStartedAtMs = millis();
        disconnectSince = wifiConnectStartedAtMs;
        lastKick = wifiConnectStartedAtMs;
        bootGraceHandled = true;
      }
    }
    return;
  }

  // If no credentials are saved, go straight to config portal.
  if (!wm.getWiFiIsSaved()) {
    startPortalNonBlocking(false);
    return;
  }

  if (!bootGraceHandled) {
    if (millis() - disconnectSince < BOOT_GRACE_MS) {
      return;
    }
    bootGraceHandled = true;
  }

  kickReconnectWindow();
}

static void wifiTask(void *arg) {
  (void)arg;
  while (true) {
    wifiService();
    vTaskDelay(WIFI_TASK_PERIOD_TICKS);
  }
}

static void cloudService() {
  // Do not publish default globals while setup is still loading saved settings.
  if (!cloudStateReady || !wifiOk()) {
    return;
  }

  unsigned long now = millis();
  if (now - lastCloudRequestMs < CLOUD_HTTP_INTERVAL_MS) {
    return;
  }
  lastCloudRequestMs = now;

  StaticJsonDocument<CLOUD_TX_JSON_CAPACITY> doc;
  // down and the local temperature fields are also UI edit buffers. While a local
  // settings edit is pending, report the pre-edit values until the save block
  // has finished. Outside editing, down remains live so countdown updates work.
  uint32_t reportedSessionTime = cloudLocalSettingsSavePending ? cloudSessionBeforeEdit : down;
  // Use the saved Fahrenheit copies for cloud reporting. The live display
  // helpers can be recalculated from Celsius when the UI is in C mode, which
  // would lose exact app values such as 177F -> 80C -> 176F.
  int reportedTargetTemperature = cloudLocalSettingsSavePending
                                  ? cloudTempBeforeEdit
                                  : ((steamerOff == 0) ? save_fahrenheit : save_combifahrenheit);
  int reportedActualTemperature = ((9 * (int)actualTemperature_data) / 5) + 32;
  doc["id"] = deviceId;
  doc["temperature"] = reportedActualTemperature;
  doc["targetTemperature"] = reportedTargetTemperature;
  doc["maxTemp"] = temperatureSet_data;
  doc["sessionTime"] = reportedSessionTime;
  doc["heaterOn"] = heaterOn;
  doc["lightsOn"] = lightActive;
  doc["temperatureScale"] = temperatureScale;
  doc["apiKey"] = apiKey;

  JsonArray errors = doc.createNestedArray("errors");
  if (ERROR1_data >= 1 && ERROR1_data <= 10) {
    char errorCode[4];
    snprintf(errorCode, sizeof(errorCode), "E%u", ERROR1_data);
    errors.add(errorCode);
  }

  String payload;
  serializeJson(doc, payload);

  HTTPClient localHttp;
  localHttp.begin(urlString);
  localHttp.addHeader("Content-Type", "application/json");
  localHttp.addHeader("API-Key", apiKey);
  localHttp.setTimeout(700);

  int httpResponseCode = localHttp.PUT(payload);
  // Serial.print("[HTTP] Response code: ");
  // Serial.println(httpResponseCode);

  if (httpResponseCode == 200) {
    StaticJsonDocument<CLOUD_RX_JSON_CAPACITY> responseDoc;
    DeserializationError err = deserializeJson(responseDoc, localHttp.getStream());
    if (err) {
      // Serial.println(err.c_str());
    } else {
      JsonObject commands = responseDoc["commands"];
      if (!commands.isNull()) {
        if (commands.containsKey("setSessionTime")) {
          uint32_t incomingSession = commands["setSessionTime"].as<uint32_t>();
          // Queue only: processDbPersistQueue() validates and saves this value
          // before it changes the live value that is reported back to the cloud.
#if defined(ARDUINO_ARCH_ESP32)
          portENTER_CRITICAL(&dbPersistMux);
          dbSessionSaveSeconds = incomingSession;
          dbSessionSavePending = true;
          portEXIT_CRITICAL(&dbPersistMux);
#else
          dbSessionSaveSeconds = incomingSession;
          dbSessionSavePending = true;
#endif
        }

static inline void otaResetTriggerSequence() {
  otaTapWindowStartMs = 0;
  otaTapCount = 0;
  otaTriggerArmed = false;
  otaArmedAtMs = 0;
  otaComboHoldStartMs = 0;
}

static bool otaHandleTriggerCombo() {
  if (powerbtn == LOW && upbtn == LOW) {
    otaComboButtonLockout = false;
  }

  if (otaInProgress) {
    otaResetTriggerSequence();
    return false;
  }

  if (buttonOnPressed != 1) {
    otaResetTriggerSequence();
    return false;
  }

  if (!otaTriggerArmed) {
    return otaComboButtonLockout && powerbtn == HIGH;
  }

  if (upbtn == HIGH) {
    otaComboButtonLockout = true;
  }

  if (otaBlockedByNetworkError()) {
    otaResetTriggerSequence();
    return otaComboButtonLockout && powerbtn == HIGH;
  }

  unsigned long now = millis();
  if (otaArmedAtMs == 0 || (now - otaArmedAtMs) > OTA_ARM_WINDOW_MS) {
    otaResetTriggerSequence();
    return otaComboButtonLockout && powerbtn == HIGH;
  }

  bool comboHeld = (powerbtn == HIGH && upbtn == HIGH);
  if (!comboHeld) {
    otaComboHoldStartMs = 0;
    return false;
  }

  if (otaComboHoldStartMs == 0) {
    otaComboHoldStartMs = now;
    Serial.println("OTA combo started: keep holding POWER + UP...");
    return true;
  }

  if ((now - otaComboHoldStartMs) >= OTA_COMBO_HOLD_MS) {
    otaTriggerPending = true;
    otaTriggerPendingSinceMs = now;
    otaFeedbackBeepMs = OTA_TRIGGERED_BEEP_MS;
    otaResetTriggerSequence();
    Serial.println("OTA trigger queued.");
  }

  return true;
}

static bool otaShouldCheck() {
  if (otaInProgress) return false;
  if (!otaTriggerPending) return false;
  if (buttonOnPressed != 1) return false;  // only allow OTA trigger while in OFF mode
  if (!wifiOk()) return false;
  if (portalActive) return false;
  if (otaBlockedByNetworkError()) return false;
  if (OTA_FIRMWARE_URL == nullptr || OTA_FIRMWARE_URL[0] == '\0') return false;
  return true;
}

static void otaService() {
  if (!otaShouldCheck()) {
    if (otaTriggerPending &&
        otaTriggerPendingSinceMs != 0 &&
        (millis() - otaTriggerPendingSinceMs) >= OTA_PENDING_TIMEOUT_MS) {
      otaTriggerPending = false;
      otaTriggerPendingSinceMs = 0;
      otaProgressPercent = -1;
      otaFeedbackBeepMs = OTA_ARMED_BEEP_MS;
      Serial.println("OTA: pending canceled because OTA could not start.");
    }
    return;
  }

  otaInProgress = true;
  otaTriggerPending = false;
  otaTriggerPendingSinceMs = 0;
  otaProgressPercent = 0;

  Serial.printf("OTA: current version=%s\n", CURRENT_FIRMWARE_VERSION);
  if (!otaDownloadAndApply(String(OTA_FIRMWARE_URL))) {
    Serial.println("OTA: update failed");
  }

  otaInProgress = false;
}

static bool otaDownloadAndApply(const String &assetUrl) {
  WiFiClientSecure client;
  client.setInsecure();

  HTTPClient http;
  http.setFollowRedirects(HTTPC_STRICT_FOLLOW_REDIRECTS);

  if (!http.begin(client, assetUrl)) return false;

  http.setTimeout(15000);
  http.addHeader("Accept", "application/octet-stream");
  http.setUserAgent("ESP32-OTA-Client");

  int httpCode = http.GET();
  if (httpCode != HTTP_CODE_OK) {
    Serial.printf("OTA: asset download HTTP %d\n", httpCode);
    http.end();
    return false;
  }

  int contentLength = http.getSize();
  Serial.printf("OTA: download size=%d\n", contentLength);

  if (!Update.begin(contentLength > 0 ? contentLength : UPDATE_SIZE_UNKNOWN)) {
    Serial.println("OTA: Update.begin() failed");
    http.end();
    return false;
  }
  otaProgressPercent = 0;

  WiFiClient *stream = http.getStreamPtr();

  static uint8_t buffer[1024];
  size_t totalWritten = 0;
  int lastPercent = -1;

  unsigned long lastData = millis();

  while (true) {
    size_t available = stream->available();

    if (available) {
      size_t toRead = std::min((size_t)available, sizeof(buffer));
      int len = stream->readBytes(buffer, toRead);
      if (len <= 0) {
        Serial.println("OTA: stream read timeout/error");
        Update.abort();
        http.end();
        otaProgressPercent = -1;
        return false;
      }

      size_t written = Update.write(buffer, (size_t)len);
      if (written != (size_t)len) {
        Serial.printf("OTA: flash write mismatch (in=%d out=%u)\n", len, (unsigned)written);
        Update.abort();
        http.end();
        otaProgressPercent = -1;
        return false;
      }
      totalWritten += written;
      lastData = millis();

      if (contentLength > 0) {
        int percent = (totalWritten * 100) / (size_t)contentLength;
        if (percent < 1) {
          percent = 1;
        } else if (percent > 100) {
          percent = 100;
        }

        if (percent != lastPercent) {
          Serial.printf("OTA Progress: %d%%\n", percent);
          lastPercent = percent;
          otaProgressPercent = percent;
        }
      }
    } else {
      // exit if done OR timeout
      if ((contentLength > 0 && totalWritten >= (size_t)contentLength) || !http.connected()) {
        break;
      }
      if (millis() - lastData > 15000) {
        Serial.println("OTA: timed out waiting for more data");
        Update.abort();
        http.end();
        otaProgressPercent = -1;
        return false;
      }
      delay(1);
    }
  }

  if (contentLength > 0 && totalWritten != (size_t)contentLength) {
    Serial.printf("OTA: incomplete payload (%u/%u)\n", (unsigned)totalWritten, (unsigned)contentLength);
    Update.abort();
    http.end();
    otaProgressPercent = -1;
    return false;
  }

  if (!Update.end()) {
    Serial.printf("OTA: Update.end() failed, err=%d\n", Update.getError());
    Update.abort();
    http.end();
    otaProgressPercent = -1;
    return false;
  }

  if (!Update.isFinished()) {
    Serial.println("OTA: image not finished");
    Update.abort();
    http.end();
    otaProgressPercent = -1;
    return false;
  }

  http.end();

  otaProgressPercent = 100;
  ERROR1_data = 0;
  ERROR2_data = 0;
  errorTrue = false;
  errorClearPending = false;
  errorDisplayAcknowledged = false;
  activatederror = 0;
  displayBlink2 = 0;
  eepromWriteByte(EEPROM_ADDR_SAVED_POWER_ERROR, 0);
  Serial.println("OTA Complete!");
  delay(2000);

  otaRestartPending = true;
  displayBlankFrameOps();
  delay(150);
  ESP.restart();
  return true;
}

static void processDbPersistQueue() {
  bool applySession = false;
  uint32_t sessionSeconds = 0;
  bool applyTemp = false;
  int16_t tempValue = 0;
  bool didDbSave = false;

#if defined(ARDUINO_ARCH_ESP32)
  portENTER_CRITICAL(&dbPersistMux);
  if (dbSessionSavePending) {
    applySession = true;
    sessionSeconds = dbSessionSaveSeconds;
    dbSessionSavePending = false;
  }
  if (dbTempSavePending) {
    applyTemp = true;
    tempValue = dbTempSaveValue;
    dbTempSavePending = false;
  }
  portEXIT_CRITICAL(&dbPersistMux);
#else
  if (dbSessionSavePending) {
    applySession = true;
    sessionSeconds = dbSessionSaveSeconds;
    dbSessionSavePending = false;
  }
  if (dbTempSavePending) {
    applyTemp = true;
    tempValue = dbTempSaveValue;
    dbTempSavePending = false;
  }
#endif

  if (applySession) {
    uint32_t maxAllowed = (maxTime >= 300) ? maxTime : 300;
    if (sessionSeconds < 300) sessionSeconds = 300;
    if (sessionSeconds > maxAllowed) sessionSeconds = maxAllowed;

    // Save first, then expose the new session to the running device/cloud state.
    uint8_t savedHours = (sessionSeconds / 3600) % 100;
    uint8_t savedMinutes = (sessionSeconds / 60) % 60;
    eepromWriteByte(6, savedHours);
    eepromWriteByte(7, savedMinutes);
    eepromWriteByte(23, savedHours);
    eepromWriteByte(24, savedMinutes);

    time_On2 = sessionSeconds;
    down = sessionSeconds;
    hh = savedHours;
    mm = savedMinutes;
    hh_session = savedHours;
    mm_session = savedMinutes;
    savePoints[0] = 0;
    didDbSave = true;
  }

  if (applyTemp) {
    int requestedFahrenheit = tempValue;
    // The cloud/mobile contract is Fahrenheit. Convert once at the boundary
    // where the power controller and persisted Celsius setpoint are updated.
    int setpoint = ((requestedFahrenheit - 32) * 5) / 9;
    if (setpoint < 30) setpoint = 30;
    if (setpoint > temperatureSet_data) setpoint = temperatureSet_data;

    int savedFahrenheit = requestedFahrenheit;
    if (setpoint == 30 && requestedFahrenheit < (((9 * setpoint) / 5) + 32)) {
      savedFahrenheit = ((9 * setpoint) / 5) + 32;
    }
    if (setpoint == temperatureSet_data && requestedFahrenheit > (((9 * setpoint) / 5) + 32)) {
      savedFahrenheit = ((9 * setpoint) / 5) + 32;
    }
    // Persist the accepted value before changing the live controller/cloud value.
    eepromWriteByte(3, setpoint);
    eepromWriteByte(14, savedFahrenheit);

    tempSet_on_data = setpoint;
    tempSet_on = setpoint;
    save_tempSet_on = tempSet_on;
    fahrenheit = savedFahrenheit;
    save_fahrenheit = fahrenheit;
    // Refresh the Fahrenheit menu/display buffers too. Without this, an app
    // update can save correctly but the next menu press may still show the old
    // Fahrenheit value that was cached before the cloud command arrived.
    tempSet_on_fah = savedFahrenheit;
    if (bothChange == 1 && menu_On_setttings == 0) {
      displayForTemp = tempSet_on_fah;
    }

    ICSC.send(5, 'k', 5, (char *)&tempSet_on_data);  // send app-updated temp to power controller
    didDbSave = true;
  }

  if (didDbSave) {
    pzDetect = 1;
    intervalBuz = 300;
  }
}

void buttonFunction() {
  byte rawMenu = LOW;
  byte rawPower = LOW;
  byte rawDown = LOW;
  byte rawUp = LOW;

#if defined(ARDUINO_ARCH_ESP32)
  // Keep scan atomic so Wi-Fi/background tasks cannot preempt while shared pins are input.
  portENTER_CRITICAL(&buttonScanMux);
  allColumnsOffFast();
  sharedButtonPinsToInputFast();
  // Small settle window before sampling.
  delayMicroseconds(BUTTON_INPUT_SETTLE_US);

  uint32_t inBits = REG_READ(GPIO_IN_REG);
  rawMenu = ((inBits & (1UL << 6)) == 0) ? HIGH : LOW;   // IO6
  rawPower = ((inBits & (1UL << 1)) == 0) ? HIGH : LOW;  // IO1
  rawDown = ((inBits & (1UL << 7)) == 0) ? HIGH : LOW;   // IO7
  rawUp = ((inBits & (1UL << 0)) == 0) ? HIGH : LOW;     // IO0

  sharedButtonPinsToOutputHighFast();
  portEXIT_CRITICAL(&buttonScanMux);
#else
  noDisplay7Seg();
  pinMode(7, INPUT_PULLUP);
  pinMode(6, INPUT_PULLUP);
  pinMode(1, INPUT_PULLUP);
  pinMode(0, INPUT_PULLUP);
  rawMenu = (digitalRead(6) == LOW) ? HIGH : LOW;   // menu button
  rawPower = (digitalRead(1) == LOW) ? HIGH : LOW;  // power button
  rawDown = (digitalRead(7) == LOW) ? HIGH : LOW;   // minus button
  rawUp = (digitalRead(0) == LOW) ? HIGH : LOW;     // plus button
  pinMode(7, OUTPUT);
  pinMode(6, OUTPUT);
  pinMode(1, OUTPUT);
  pinMode(0, OUTPUT);
  digitalWrite(7, HIGH);
  digitalWrite(6, HIGH);
  digitalWrite(1, HIGH);
  digitalWrite(0, HIGH);
#endif

  unsigned long nowMs = millis();
  menubtn = debounceButtonState(rawMenu, menuBtnDebounce, nowMs);
  powerbtn = debounceButtonState(rawPower, powerBtnDebounce, nowMs);
  downbtn = debounceButtonState(rawDown, downBtnDebounce, nowMs);
  upbtn = debounceButtonState(rawUp, upBtnDebounce, nowMs);
}

      buttonActiveLow = true;
      longPressTimeLow = millis();
    }
    if ((millis() - longPressTimeLow) > longPressTime2 && no_button_work == 0) {
      longPressActiveLow = true;
      //
      unsigned long fastCount = millis();
      if (fastCount - prevfastCount >= debfastCount) {
        prevfastCount = fastCount;
        fastCnt++;
        if (fastCnt > 35) {
          fastCnt = 35;
        }
        if (fastCnt >= 0 && fastCnt < 6) {
          debfastCount = 250;
        } else if (fastCnt >= 6 && fastCnt < 20) {
          debfastCount = 175;
        } else if (fastCnt >= 20 && fastCnt < 35) {
          debfastCount = 100;
        } else debfastCount = 50;
        data--;
        if (buttonOnPressed == 2) {
          if (menu_On_setttings == 2) {
            data = data - 300;
          }
        }
      }
    }
  } else {
    if (buttonActiveLow == true) {
      //
      if (longPressActiveLow == true) {
        longPressActiveLow = false;
        fastCnt = 0;
        debfastCount = 250;
      } else {
        data--;
        if (buttonOnPressed == 2) {
          if (menu_On_setttings == 2) {
            data = data - 300;
          }
        }
      }
      buttonActiveLow = false;
    }
  }
  return data;
}

inline byte normalizeMenuSetting(byte setting) {
  if (setting > 4) setting = 0;

  // Skip unavailable humidity page.
  if (setting == 1 && !(steamerOff == 1 || humidityData_data != 0)) {
    setting = 2;
  }

  // Skip unavailable steamer/fan pages without rendering intermediate invalid pages.
  if (setting == 3 && !(combi_data == 1 && combiDetect_data != 3)) {
    setting = (fanDetect_data == 1) ? 4 : 0;
  }
  if (setting == 4 && fanDetect_data != 1) {
    setting = 0;
  }

  return setting;
}

double Thermister(double RawADC) {
  double Temp;
  Temp = 10000 * (RawADC * (5 / 1023.0)) / (5 - (RawADC * (5 / 1023.0)));
  Temp = 1 / (0.001138177396 + 0.0002326061337 * log(Temp) + 0.0000000961782205 * log(Temp) * log(Temp) * log(Temp));
  Temp = Temp - 273;
  return Temp;
}

void off() {
  ICSC.process();
  if (buttonOnPressed == 1) {  //off
    forTemp = HumidSet_read;
    forCombi = combiPLUS;
  } else {  //on
    forTemp = humidSet_on;
    forCombi = combiPLUS2;
    tenSecondsSendMillis = millis();
  }
  //formula
  if (tempDATA == 0) {
    //formula of the curve
    //for temperature settings
    if (forTemp > 59.6 && forTemp <= 110.0) {
      y = 100.0 - sqrt((1635.0 / 13.0) * (forTemp - 30.0));
    } else if (forTemp > 52.579 && forTemp <= 59.6) {
      y = 134.35 - 1.6 * forTemp;
    } else if (forTemp > 45.0 && forTemp <= 52.579) {
      y = 100.0 - sqrt((4250.0 / 13.0) * (forTemp - 45.0));
    } else {
      y = 100.0;
    }
    //for humidity settings
    if (forCombi >= 0.0 && forCombi <= 38.98) {
      x2 = 30.0 + (((13.0) * (100.0 - forCombi) * (100.0 - forCombi)) / 1635.0);
    } else if (forCombi > 38.98 && forCombi < 50.224) {
      x2 = (134.35 - forCombi) / 1.6;
    } else {
      x2 = ((13.0) * (100.0 - forCombi) * (100.0 - forCombi)) / 4250.0 + 45.0;
    }
    //end of the formula
  } else {
    //formula of the curve
    //for temperature settings
    if (forTemp > 69.7 && forTemp <= 120.0) {
      y = 100.0 - sqrt((1635.0 / 13.0) * (forTemp - 40.0));
    } else if (forTemp > 62.531 && forTemp <= 69.7) {
      y = 150.43 - 1.6 * forTemp;
    } else if (forTemp > 55.0 && forTemp <= 62.531) {
      y = 100.0 - sqrt((4250.0 / 13.0) * (forTemp - 55.0));
    } else {
      y = 100.0;
    }
    //for humidity settings
    if (forCombi >= 0.0 && forCombi <= 38.83) {
      x2 = 40.0 + (((13.0) * (100.0 - forCombi) * (100.0 - forCombi)) / 1635.0);
    } else if (forCombi > 38.83 && forCombi < 50.38) {
      x2 = (150.43 - forCombi) / 1.6;
    } else {
      x2 = ((13.0) * (100.0 - forCombi) * (100.0 - forCombi)) / 4250.0 + 55.0;
    }
    //end of the formula
  }
  bool showPowerError = errorTrue && (!errorDisplayAcknowledged || ERROR1_data == 7);
  switch (buttonOnPressed) {
    case 1:  /////////////////////////////////////////////////////////////////OFF
      if (save_tempSet_on > temperatureSet_data) {
        save_tempSet_on = temperatureSet_data;
        save_fahrenheit = celToFah(save_tempSet_on);
      }
      if (save_humidSet_on > humidLimit) {
        save_humidSet_on = humidLimit;
        save_combifahrenheit = celToFah(save_humidSet_on);
      }
      if (preRunSet_running != lastpreRunSet_running) {
        ICSC.send(5, 'P', 2, (char *)&preRunSet_running);
        lastpreRunSet_running = preRunSet_running;
      }
      savePoint = 0;
      everyOnsend = true;
      // oneTimeOnly = 1;
      // displayForView = 0;
      prevdurationSetTemp = 0;
      prevfiveMinutes = 0;  //========================5 minutes
      delaySendmillis = 0;
      maxTimeDown = maxTime;  //maximum time as reference for the session time

      if (time_On2 > maxTime)
        time_On2 = maxTime;
      else
        time_On2 = hh * 3600 + mm * 60;

      // changeDisplayTemp = false;
      setTempDisplay_On = 0;
      menu_On = false;
      if (showPowerError == false) {
        if (preRunSet == 0 && preRunSet_running == 0)  ///////////////////////////////////////////////////////// /*=================OFF state================*/
        {
          if (fanOff == 1) {
            displayLed(14);
            if (fanTimeMillis > 14400)
              fanOff = 0;
          } else
            fanTimeMillis = 0;
          fanButton();

          if (remoteOn_data == 0 && ERROR1_data == 9) {

            kas_a_lang = false;
          }
          displayBlink2 = 0;
          menu_On_setttings = 0;
          if (combiDetect_data == 2) {
            sensorChange = 2;

            sensorChange = 1;

#if defined(ARDUINO_ARCH_ESP32)
static inline uint8_t displayDigitSegments(uint8_t x) {
  switch (x) {
    case 0: return SEG_A | SEG_B | SEG_C | SEG_D | SEG_E | SEG_F;
    case 1: return SEG_B | SEG_C;
    case 2: return SEG_A | SEG_B | SEG_D | SEG_E | SEG_G;
    case 3: return SEG_A | SEG_B | SEG_C | SEG_D | SEG_G;
    case 4: return SEG_B | SEG_C | SEG_F | SEG_G;
    case 5: return SEG_A | SEG_C | SEG_D | SEG_F | SEG_G;
    case 6: return SEG_A | SEG_C | SEG_D | SEG_E | SEG_F | SEG_G;
    case 7: return SEG_A | SEG_B | SEG_C;
    case 8: return SEG_A | SEG_B | SEG_C | SEG_D | SEG_E | SEG_F | SEG_G;
    case 9: return SEG_A | SEG_B | SEG_C | SEG_D | SEG_F | SEG_G;
    default: return 0;
  }
}

static inline uint8_t displayPositiveMod10(int value) {
  int mod = value % 10;
  if (mod < 0) mod += 10;
  return (uint8_t)mod;
}

static inline void displayFrameDigit(uint8_t frame[DISPLAY_REFRESH_COLS], uint8_t col, int digit) {
  if (col < DISPLAY_REFRESH_COLS) {
    frame[col] |= displayDigitSegments(displayPositiveMod10(digit));
  }
}

static inline void displayFrameSegments(uint8_t frame[DISPLAY_REFRESH_COLS], uint8_t col, uint8_t segs) {
  if (col < DISPLAY_REFRESH_COLS) {
    frame[col] |= segs;
  }
}

static inline void displayFrameLed(uint8_t frame[DISPLAY_REFRESH_COLS], byte toDisplayled) {
  switch (toDisplayled) {
    case 11: displayFrameSegments(frame, 4, SEG_A); break;
    case 12: displayFrameSegments(frame, 4, SEG_B); break;
    case 13: displayFrameSegments(frame, 4, SEG_C); break;
    case 14: displayFrameSegments(frame, 4, SEG_D); break;
    case 15: displayFrameSegments(frame, 4, SEG_E); break;
    case 16: displayFrameSegments(frame, 4, SEG_F); break;
    case 33: displayFrameSegments(frame, 4, SEG_G); break;
    default: break;
  }
}

static void displayRenderOpToFrame(const DisplayOp &op, uint8_t frame[DISPLAY_REFRESH_COLS]) {
  int value = op.arg;
  switch (op.type) {
    case DISP_OP_NUMBER3:
      displayFrameDigit(frame, 1, value / 100);
      displayFrameDigit(frame, 2, (value % 100) / 10);
      displayFrameDigit(frame, 3, value % 10);
      break;
    case DISP_OP_OFF:
      displayFrameSegments(frame, 1, SEG_A | SEG_B | SEG_C | SEG_D | SEG_E | SEG_F);
      displayFrameSegments(frame, 2, SEG_A | SEG_E | SEG_F | SEG_G);
      displayFrameSegments(frame, 3, SEG_A | SEG_E | SEG_F | SEG_G);
      break;
    case DISP_OP_S:
      displayFrameDigit(frame, 0, value / 10);
      displayFrameDigit(frame, 1, value % 10);
      if (combi_data == 1) displayFrameDigit(frame, 3, 5);
      break;
    case DISP_OP_COLON:
      displayFrameSegments(frame, 2, SEG_DP);
      displayFrameSegments(frame, 3, SEG_DP);
      break;
    case DISP_OP_CEL:
      displayFrameSegments(frame, 1, SEG_A | SEG_D | SEG_E | SEG_F);
      displayFrameSegments(frame, 2, SEG_A | SEG_D | SEG_E | SEG_F | SEG_G);
      displayFrameSegments(frame, 3, SEG_D | SEG_E | SEG_F);
      break;
    case DISP_OP_HOT:
      displayFrameSegments(frame, 1, SEG_B | SEG_C | SEG_E | SEG_F | SEG_G);
      displayFrameSegments(frame, 2, SEG_C | SEG_D | SEG_E | SEG_G);
      displayFrameSegments(frame, 3, SEG_D | SEG_E | SEG_F | SEG_G);
      break;
    case DISP_OP_FAH:
      displayFrameSegments(frame, 1, SEG_FAH_MARK);
      displayFrameSegments(frame, 2, SEG_A | SEG_B | SEG_C | SEG_E | SEG_F | SEG_G);
      displayFrameSegments(frame, 3, SEG_B | SEG_C | SEG_E | SEG_F | SEG_G);
      break;
    case DISP_OP_ON:
      displayFrameSegments(frame, 2, SEG_A | SEG_B | SEG_C | SEG_D | SEG_E | SEG_F);
      displayFrameSegments(frame, 3, SEG_C | SEG_E | SEG_G);
      break;
    case DISP_OP_MIN:
      displayFrameDigit(frame, 2, value / 10);
      displayFrameSegments(frame, 2, SEG_DP);
      displayFrameDigit(frame, 3, value % 10);
      displayFrameSegments(frame, 3, SEG_DP);
      break;
    case DISP_OP_NUMBER_FAH3:
      displayFrameDigit(frame, 0, value / 100);
      displayFrameDigit(frame, 1, (value % 100) / 10);
      displayFrameDigit(frame, 2, value % 10);
      displayFrameSegments(frame, 3, SEG_FAH_MARK);
      break;
    case DISP_OP_NUMBER_FAH2:
      displayFrameDigit(frame, 1, value / 10);
      displayFrameDigit(frame, 2, value % 10);
      displayFrameSegments(frame, 3, SEG_FAH_MARK);
      break;
    case DISP_OP_LED:
      displayFrameLed(frame, (byte)value);
      break;
    case DISP_OP_NUMBER:
      displayFrameDigit(frame, 2, value / 10);
      displayFrameDigit(frame, 3, value % 10);
      break;
    case DISP_OP_NUMBER_NO_LEAD:
      if (value >= 10) displayFrameDigit(frame, 2, value / 10);
      displayFrameDigit(frame, 3, value % 10);
      break;
    case DISP_OP_HOUR:
      if (value >= 10) displayFrameDigit(frame, 0, value / 10);
      displayFrameDigit(frame, 1, value % 10);
      break;
    case DISP_OP_REM:
      displayFrameSegments(frame, 0, SEG_E | SEG_G);
      displayFrameSegments(frame, 2, SEG_A | SEG_B | SEG_C | SEG_D | SEG_E | SEG_F);
      displayFrameSegments(frame, 3, SEG_C | SEG_E | SEG_G);
      break;
    case DISP_OP_ERROR:
      if (value >= 10) {
        displayFrameSegments(frame, 1, SEG_A | SEG_D | SEG_E | SEG_F | SEG_G);
        displayFrameDigit(frame, 2, value / 10);
        displayFrameDigit(frame, 3, value % 10);
      } else {
        displayFrameSegments(frame, 2, SEG_A | SEG_D | SEG_E | SEG_F | SEG_G);
        displayFrameDigit(frame, 3, value);
      }
      break;
    case DISP_OP_OPEN:
      displayFrameSegments(frame, 0, SEG_C | SEG_D | SEG_E | SEG_G);
      displayFrameSegments(frame, 1, SEG_A | SEG_B | SEG_E | SEG_F | SEG_G);
      displayFrameSegments(frame, 2, SEG_A | SEG_D | SEG_E | SEG_F | SEG_G);
      displayFrameSegments(frame, 3, SEG_C | SEG_E | SEG_G);
      break;
    case DISP_OP_KLOCK:
      displayFrameSegments(frame, 0, SEG_G);
      displayFrameSegments(frame, 1, SEG_G);
      displayFrameSegments(frame, 2, SEG_G);
      displayFrameSegments(frame, 3, SEG_G);
      break;
    case DISP_OP_FILL:
      displayFrameSegments(frame, 0, SEG_FAH_MARK);
      displayFrameSegments(frame, 1, SEG_B | SEG_C);
      displayFrameSegments(frame, 2, SEG_D | SEG_E | SEG_F);
      displayFrameSegments(frame, 3, SEG_D | SEG_E | SEG_F);
      break;
    case DISP_OP_DRY:
      displayFrameSegments(frame, 1, SEG_B | SEG_C | SEG_D | SEG_E | SEG_G);
      displayFrameSegments(frame, 2, SEG_E | SEG_G);
      displayFrameSegments(frame, 3, SEG_B | SEG_C | SEG_D | SEG_F | SEG_G);
      break;
    default:
      break;
  }
}

static void displayRenderOpsToFrame(const DisplayOp *ops, uint8_t cnt, uint8_t frame[DISPLAY_REFRESH_COLS]) {
  for (uint8_t i = 0; i < DISPLAY_REFRESH_COLS; i++) {
    frame[i] = 0;
  }
  for (uint8_t i = 0; i < cnt; i++) {
    displayRenderOpToFrame(ops[i], frame);
  }
}

static void displayTask(void *arg) {
  (void)arg;
  DisplayOp localOps[DISPLAY_OP_CAP];
  uint8_t localFrame[DISPLAY_REFRESH_COLS];
  while (true) {
    if (!displayReplayEnabled) {
      vTaskDelay(1);
      continue;
    }

    uint8_t cnt;
    portENTER_CRITICAL(&displayOpsMux);
    cnt = displayActiveCount;
    if (cnt > DISPLAY_OP_CAP) cnt = DISPLAY_OP_CAP;
    for (uint8_t i = 0; i < cnt; i++) {
      localOps[i] = displayActiveOps[i];
    }
    portEXIT_CRITICAL(&displayOpsMux);

    if (cnt == 0) {
      displayClearRefreshFrame();
      vTaskDelay(1);
      continue;
    }

    displayRenderOpsToFrame(localOps, cnt, localFrame);
    displayCommitRefreshFrame(localFrame);
    vTaskDelay(1);
  }
}

     if (VERSION_data != 0) {

      lightStop = 1;

    if (steamerOff != laststeamerOff) {
      ICSC.send(5, 'C', 5, (char *)&steamerOff);
      if (no40min == 0) {
        if (steamerOff == 1) {
          if (DHP != 3)
            maxTimeDown = maxTimeDown - 2400;

          addTime = 1;
        }
      }
      tempSet_on_data = (steamerOff == 0) ? tempSet_on : humidSet_on;
      ICSC.send(5, 'k', 5, (char *)&tempSet_on_data);
      laststeamerOff = steamerOff;
    }
  }

  if (combiDetect_data == 3) {
    steamerOff = 0;
    steamerOff_PRERUN = 0;
  }

  if (HOT_data == 0)
    steamerMillis = 0;

  if (fastbeep3 == 1) {
    fastBeeping2();
    fastbeep3 = 0;
  }
  displayCommitFrameOps();
  delay(10);
}
